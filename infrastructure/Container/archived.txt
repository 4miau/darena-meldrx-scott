        private AzureNative.App.ContainerApp createContainerApp(string appName, ProjectConfig p, AzureNative.OperationalInsights.Workspace log, Output<string> logKey, AzureNative.ContainerRegistry.Registry container_registry, AzureNative.App.ManagedEnvironment container_managed_env, Output<string> acrKey, ResourceGroup rg)
    {
        var random_acr_uuid = new Random.RandomUuid("random_acr_uuid");

        var acr_role = new AzureNative.Authorization.RoleAssignment("roleAssignment-acr", new()
        {
            PrincipalId = p.acrClientId,
            PrincipalType = "ServicePrincipal",
            RoleAssignmentName = random_acr_uuid.Result,
            RoleDefinitionId = "/subscriptions/" + p.subscriptionId + "/providers/Microsoft.Authorization/roleDefinitions/b24988ac-6180-42a0-ab88-20f7382dd24c",
            Scope = container_registry.Id,
        });

        return new AzureNative.App.ContainerApp($"container-app-{appName}", new()
        {
            ContainerAppName = $"container-app-{appName}-{p.projectName}-{p.env}",
            ResourceGroupName = rg.Name,
            Configuration = new AzureNative.App.Inputs.ConfigurationArgs
            {
                Secrets = new AzureNative.App.Inputs.SecretArgs
                {
                    Name = container_registry.Name.Apply(v => v),
                    Value = acrKey
                },
                Registries = new AzureNative.App.Inputs.RegistryCredentialsArgs
                {
                    Username = container_registry.Name.Apply(v => v),
                    PasswordSecretRef = container_registry.Name.Apply(v => v),
                    //Identity = user_acrpull.Id,
                    Server = container_registry.LoginServer.Apply(v => v),
                },
                Ingress = new AzureNative.App.Inputs.IngressArgs
                {
                    AllowInsecure = false,
                    Transport = "Auto",
                    External = true,
                    TargetPort = 443,
                },
            },
            ManagedEnvironmentId = container_managed_env.Id,
            Template = new AzureNative.App.Inputs.TemplateArgs
            {
                Containers = new[]
                {
                    new AzureNative.App.Inputs.ContainerArgs
                    {
                        Name = $"container-app-{appName}-{p.projectName}-{p.env}",
                        Image = container_registry.LoginServer.Apply(v => $"{v}/jobs:latest"),
                        Probes = new[]
                        {
                            new AzureNative.App.Inputs.ContainerAppProbeArgs
                            {
                                HttpGet = new AzureNative.App.Inputs.ContainerAppProbeHttpGetArgs
                                {
                                    Path = "/healthz",
                                    Port = 443,
                                },
                                InitialDelaySeconds = 3,
                                PeriodSeconds = 3,
                                Type = "Liveness",
                            },
                        },
                    },
                },
                Scale = new AzureNative.App.Inputs.ScaleArgs
                {
                    MaxReplicas = 20,
                    MinReplicas = 1,
                    Rules = new[]
                    {
                        new AzureNative.App.Inputs.ScaleRuleArgs
                        {
                            Name = "memory-scaling",
                            Custom = new AzureNative.App.Inputs.CustomScaleRuleArgs
                            {
                                Metadata =
                                {
                                    { "Utilization", "80" },
                                },
                                Type = "memory",
                            },
                        },
                    },
                },
            },
            Tags = { { "env", p.env } }
        });
    }
